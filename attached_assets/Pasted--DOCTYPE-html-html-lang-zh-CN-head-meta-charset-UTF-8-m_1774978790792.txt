                </div>
                <button onclick="refreshData()" 
                        class="px-6 py-2.5 bg-blue-600 text-white rounded-2xl hover:bg-blue-700 transition-colors">

                <div class="text-amber-600 text-sm font-medium">待审核</div>
                <div id="pending-count" class="text-4xl font-bold text-slate-800 mt-2">0</div>
            </div>
            <div class="bg-white rounded-3xl p-6 shadow-sm border border-slate-100">

            <div class="flex justify-between items-center mb-6">

                <input id="search-input" type="text" placeholder="搜索项目名称或事件..." 
                       class="px-5 py-2.5 border border-slate-300 rounded-2xl focus:outline-none focus:border-blue-500 w-80">

    <script>
        let claims = [
            {
                id: 1,
                eventTitle: "Colosseum Frontier 黑客松",
                projectName: "Solana 中文社区",
                contact: "@solana_cn",
                reason: "希望作为社区方提供中文直播、开发者招募支持以及线下 Meetup 组织。",
                eventLink: "#",
                status: "pending",
                submitTime: "2026-04-01 14:30"
            },
            {
                id: 2,
                eventTitle: "Solana Accelerate USA",
                projectName: "Ocean Ventures",
                contact: "hello@ocean.ventures",
                reason: "希望作为赞助方参与线下活动，并提供投资人资源对接。",
                eventLink: "#",
                status: "approved",
                submitTime: "2026-03-28 09:15"
            }
        ];

        function getStatusClass(status) {
            if (status === 'pending') return 'status-pending';
            if (status === 'approved') return 'status-approved';
            return 'status-rejected';
        }

        function getStatusText(status) {
            if (status === 'pending') return '待审核';
            if (status === 'approved') return '已通过';
            return '已拒绝';
        }

        function renderClaims(filteredClaims) {
            const container = document.getElementById('claims-list');
            container.innerHTML = '';

            document.getElementById('total-count').textContent = claims.length;

            const pending = claims.filter(c => c.status === 'pending').length;
            const approved = claims.filter(c => c.status === 'approved').length;
            const rejected = claims.filter(c => c.status === 'rejected').length;

            document.getElementById('pending-count').textContent = pending;
            document.getElementById('approved-count').textContent = approved;
            document.getElementById('rejected-count').textContent = rejected;

            filteredClaims.forEach(claim => {
                const html = `
                    <div class="bg-white rounded-3xl p-7 shadow-sm border border-slate-100">
                        <div class="flex justify-between items-start">
                            <div class="flex-1">
                                <h3 class="font-semibold text-lg">${claim.eventTitle}</h3>
                                <p class="text-slate-600 mt-1">申请方：${claim.projectName}</p>
                                <p class="text-sm text-slate-500 mt-2">联系方式：${claim.contact}</p>
                            </div>
                            <div class="text-right">
                                <span class="px-5 py-1.5 rounded-2xl text-sm font-medium ${getStatusClass(claim.status)}">
                                    ${getStatusText(claim.status)}
                                </span>
                                <p class="text-xs text-slate-400 mt-3">${claim.submitTime}</p>
                            </div>
                        </div>

                        <div class="mt-5 text-sm text-slate-600 bg-slate-50 p-5 rounded-2xl">
                            <strong>认领理由：</strong><br>
                            ${claim.reason}
                        </div>

                        <div class="mt-6 flex gap-4">
                            <a href="${claim.eventLink}" target="_blank" 
                               class="text-blue-600 hover:underline text-sm">查看原事件 →</a>
                            
                            ${claim.status === 'pending' ? `
                                <button onclick="approveClaim(${claim.id})" 
                                        class="px-6 py-2 bg-emerald-600 text-white text-sm rounded-2xl hover:bg-emerald-700">通过</button>
                                <button onclick="rejectClaim(${claim.id})" 
                                        class="px-6 py-2 bg-red-600 text-white text-sm rounded-2xl hover:bg-red-700">拒绝</button>
                            ` : ''}
                        </div>
                    </div>
                `;
                container.innerHTML += html;
            });
        }

        function approveClaim(id) {
            const claim = claims.find(c => c.id === id);
            if (claim) {
                claim.status = 'approved';
                alert(`✅ 已通过 ${claim.projectName} 的认领申请`);
                renderClaims(claims);
            }
        }

        function rejectClaim(id) {
            const claim = claims.find(c => c.id === id);
            if (claim) {
                claim.status = 'rejected';
                alert(`❌ 已拒绝 ${claim.projectName} 的认领申请`);
                renderClaims(claims);
            }
        }

        function refreshData() {
            alert("实际项目中这里会从 Supabase / 后端拉取最新数据");
            renderClaims(claims);
        }

        // 简单搜索
        document.getElementById('search-input').addEventListener('input', (e) => {
            const term = e.target.value.toLowerCase();
            const filtered = claims.filter(c => 
                c.projectName.toLowerCase().includes(term) || 
                c.eventTitle.toLowerCase().includes(term)
            );
            renderClaims(filtered);
        });

        window.onload = () => renderClaims(claims);
    </script>
